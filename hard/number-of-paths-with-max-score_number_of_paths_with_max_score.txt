                if (maxPrev != INT_MIN) {
                    maxSum[i][j] = maxPrev + (board[i][j] - '0');
                    count[i][j] = cnt;
                }
            }
        }
        
        if (maxSum[0][0] == INT_MIN) {
            return {0, 0};
        }
        return {maxSum[0][0], count[0][0]};
    }
};
