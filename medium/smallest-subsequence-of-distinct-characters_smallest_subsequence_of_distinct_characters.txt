            
            while (!result.empty() && result.back() > s[i] && last[result.back() - 'a'] > i) {
                seen[result.back() - 'a'] = false;
                result.pop_back();
            }
            
            result.push_back(s[i]);
            seen[s[i] - 'a'] = true;
        }
        
        return result;
    }
};
