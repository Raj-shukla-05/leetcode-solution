            right[i] = st.empty() ? n - i : st.top
            () - i;
            st.push(i);
        }
        
        long long sum = 0;
        for (int i = 0; i < n; i++) {
            sum = (sum + (long long)arr[i] * left
            [i] * right[i]) % MOD;
        }
        
        return sum;
    }
};
